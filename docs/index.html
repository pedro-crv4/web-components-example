### Implementando um custom element 
                    
                        class MyCustomElement extends HTMLElement {
                            static observedAttributes = ["color", "size"];
                        
                            constructor() {
                                // Sempre chamar o método super no construtor
                                super();
                            }
                        
                            connectedCallback() {
                                console.log("Elemento adicionado à página.");
                            }
                        
                            disconnectedCallback() {
                                console.log("Elemento removido da página.");
                            }
                        
                            adoptedCallback() {
                                console.log("Elemento movido para outra página.");
                            }
                        
                            attributeChangedCallback(name, oldValue, newValue) {
                                console.log(`Atributo ${name} mudou.`);
                            }
                        }
                          
                        customElements.define("my-custom-element", MyCustomElement);
